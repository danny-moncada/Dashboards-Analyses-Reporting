	<!---   get goal title   --->
    <cfquery name="GoalTitle" datasource=#GoalSettingDataSource#>

        select   distinct
		         case 
                     when a.um_jobcode_group in ('AA', 'AP')
                         then 'Professional and Administrative Goals'
                     when a.um_jobcode_group in ('CS')
                         then 'Civil Service Goals'
                     when a.um_jobcode_group in ('LR')
                         then 'Labor Represented Goals'
                     else 'Faculty Goals'
                 end goal_title1,
				 case 
                     when a.um_jobcode_group in ('AA', 'AP')
                         then 'SYSTEMWIDE'
                     when a.um_jobcode_group in ('CS')
                         then 'by CAMPUS'
                     when a.um_jobcode_group in ('LR')
                             then 'by CAMPUS'
                     else 'by ENTITY'
                 end goal_title2

		from     oiradm.ps_dweo_person_prim_job_evw a
						 
        where    a.um_jobcode_group in ('#PositionInfo.um_jobcode_group#')
 
    </cfquery>
			
	<cfswitch expression="#PositionInfo.um_jobcode_group#">  
        <cfcase value="AA,AP">  
		
			<!---   get goal information   --->
            <cfquery name="GoalInformation" datasource=#GoalSettingDataSource#>

                select   case 
                             when a.um_jobcode_group in ('AA', 'AP')
                                 then 'Professional and Administrative Goals SYSTEMWIDE'
                             when a.um_jobcode_group in ('CS')
                                 then 'Civil Service Goals by CAMPUS'
                             when a.um_jobcode_group in ('LR')
                                 then 'Labor Represented Goals by CAMPUS'
                             else 'Faculty Goals by ENTITY'
                         end GOALS,
						 
                         a.um_jobcode_group,
                         a.um_jobcode_group_descr,
                         a.ipedsscode||a.eeo_jobcode_group eeo_jobcode_group,
						 
                         case 
                             when a.um_jobcode_group in ('AA', 'AP', 'FA')
                                 then e.academic_title
                             else e.non_academic_title
                         end eeo_jobcode_group_descr,
						 
						 count (a.emplid) total_workforce,
						 
                         <!---     female     --->
						 
						 sum(decode (a.sex, 'F', 1, 0)) female_workforce,
                         c.fem_avail_tot as fem_avail_pct,
                         round((count (a.emplid) * c.fem_avail_tot/100),1) goal_fem,
                         case
                             when sum(decode (a.sex, 'F', 1, 0)) - round((count (a.emplid) * c.fem_avail_tot/100),1) < 0 
                                 then sum(decode (a.sex, 'F', 1, 0)) - round((count (a.emplid) * c.fem_avail_tot/100),1)
                             else 0
                         end ntg_fem,
						 
                         <!---     minority     --->
						 
						 sum(case when a.multi_re_indicator in ('2', '3', '4', '5', '7', 'H', 'M') then 1 else 0 end) as minority_workforce,
						 
						 c.min_avail_tot as min_avail_pct,
                         round((count (a.emplid)*c.min_avail_tot/100),1) goal_min,
                         case
                             when sum(case when a.multi_re_indicator in ('2', '3', '4', '5', '7', 'H', 'M') then 1 else 0 end) - round((count (a.emplid)*c.min_avail_tot/100),1) < 0
                                 then sum(case when a.multi_re_indicator in ('2', '3', '4', '5', '7', 'H', 'M') then 1 else 0 end) - round((count (a.emplid)*c.min_avail_tot/100),1)
                             else 0
                         end ntg_min,
						 
						 <!---     disabled     --->
						 
						 sum(decode (a.disabled, 'Y', 1, 0)) as disabled_workforce,
						 
						 d.disabled as dis_avail_pct,
						 round((count (a.emplid) * d.disabled/100),1) goal_dis,
						 case
                             when sum(decode (a.disabled, 'Y', 1, 0)) - round((count (a.emplid) * d.disabled/100),1) < 0 
                                 then sum(decode (a.disabled, 'Y', 1, 0)) - round((count (a.emplid) * d.disabled/100),1)
                             else 0
                         end ntg_dis,
						 
						 <!---     veteran     --->
						 
						 sum(case when (a.disabled_vet in ('Y') or a.military_status in ('4', 'A', 'F', 'O', 'R', 'S')) then 1 else 0 end) as veteran_workforce,
								  
						 d.veteran as vet_avail_pct,
						 round((count (a.emplid) * d.veteran/100),1) goal_vet,
						 case
                             when sum(case when (a.disabled_vet in ('Y') or a.military_status in ('4', 'A', 'F', 'O', 'R', 'S')) then 1 else 0 end) - round((count (a.emplid) * d.veteran/100),1) < 0
                                 then sum(case when (a.disabled_vet in ('Y') or a.military_status in ('4', 'A', 'F', 'O', 'R', 'S')) then 1 else 0 end) - round((count (a.emplid) * d.veteran/100),1)
                             else 0
                         end ntg_vet

                from     oiradm.ps_dweo_person_prim_job_evw a,
                         oirweb2.eoaa_aap2factor_data c,
						 oirweb2.eoaa_aap2factor_years d,
                         oiradm.ps_eeo_jobcode_group_tbl e

						 
                where    a.um_jobcode_group = c.um_jobcode_grp
                  and    a.ipedsscode||a.eeo_jobcode_group = c.combined_eeo_job_group(+)
                  and    a.ipedsscode = e.ipedsscode(+)
                  and    a.eeo_jobcode_group = e.eeo_jobcode_group(+)
				  and    c.goal_year = d.goal_year
				  
                  and    a.jobcode not in ('0001','0004','0007','0011')
                  and    a.deptid not in ('10098','12024','12025','12058','12192')
                  and    a.ipedsscode not in ('N')
<!---                  and    a.um_campus in ('TXXX')  --->
                  and    (a.most_recent_payroll = '#MaxMostRecentPayroll.max_most_recent_payroll#'
                             or a.empl_status in ('L', 'P', 'W'))
                  and    a.um_jobcode_group in ('#PositionInfo.um_jobcode_group#')
				  and    (a.ipedsscode||a.eeo_jobcode_group) in ('#PositionInfo.eeo_jobcode_group#')
				  
                  and    c.web_avail = 'TRUE'
<!---                  and    c.campus_code in ('TXXX') --->
                  and    c.goal_year = '#MaxGoalYear.max_goal_year#'
                  and    c.um_jobcode_grp in ('#PositionInfo.um_jobcode_group#')

                group by a.um_jobcode_group, 
                         a.ipedsscode,
                         a.eeo_jobcode_group,
                         a.um_jobcode_group_descr,
                         e.academic_title,
                         e.non_academic_title,
                         c.fem_avail_tot,
                         c.min_avail_tot,
                         d.disabled,
                         d.veteran

                order by a.um_jobcode_group, eeo_jobcode_group	 

            </cfquery>

                         end ntg_fem,
						 
                         <!---     minority     --->
						 
						 sum(case when a.multi_re_indicator in ('2', '3', '4', '5', '7', 'H', 'M') then 1 else 0 end) as minority_workforce,
						 
						 c.min_avail_tot as min_avail_pct,
                         round((count (a.emplid)*c.min_avail_tot/100),1) goal_min,
                         case
                             when sum(case when a.multi_re_indicator in ('2', '3', '4', '5', '7', 'H', 'M') then 1 else 0 end) - round((count (a.emplid)*c.min_avail_tot/100),1) < 0
                                 then sum(case when a.multi_re_indicator in ('2', '3', '4', '5', '7', 'H', 'M') then 1 else 0 end) - round((count (a.emplid)*c.min_avail_tot/100),1)
                             else 0
                         end ntg_min,
						 
						 <!---     disabled     --->
						 
						 sum(decode (a.disabled, 'Y', 1, 0)) as disabled_workforce,
						 
						 d.disabled as dis_avail_pct,
						 round((count (a.emplid) * d.disabled/100),1) goal_dis,
						 case
                             when sum(decode (a.disabled, 'Y', 1, 0)) - round((count (a.emplid) * d.disabled/100),1) < 0 
                                 then sum(decode (a.disabled, 'Y', 1, 0)) - round((count (a.emplid) * d.disabled/100),1)
                             else 0
                         end ntg_dis,
						 
						 <!---     veteran     --->
						 
						 sum(case when (a.disabled_vet in ('Y') or a.military_status in ('4', 'A', 'F', 'O', 'R', 'S')) then 1 else 0 end) as veteran_workforce,
								  
						 d.veteran as vet_avail_pct,
						 round((count (a.emplid) * d.veteran/100),1) goal_vet,
						 case
                             when sum(case when (a.disabled_vet in ('Y') or a.military_status in ('4', 'A', 'F', 'O', 'R', 'S')) then 1 else 0 end) - round((count (a.emplid) * d.veteran/100),1) < 0
                                 then sum(case when (a.disabled_vet in ('Y') or a.military_status in ('4', 'A', 'F', 'O', 'R', 'S')) then 1 else 0 end) - round((count (a.emplid) * d.veteran/100),1)
                             else 0
                         end ntg_vet

                from     oiradm.ps_dweo_person_prim_job_evw a,
                         oirweb2.eoaa_aap2factor_data c,
						 oirweb2.eoaa_aap2factor_years d,
                         oiradm.ps_eeo_jobcode_group_tbl e,
						 oiradm.ps_dwhr_um_entity_zdeptid_map f
						 
                where    a.um_jobcode_group = c.um_jobcode_grp
                  and    a.ipedsscode||a.eeo_jobcode_group = c.combined_eeo_job_group(+)
                  and    a.ipedsscode = e.ipedsscode(+)
                  and    a.eeo_jobcode_group = e.eeo_jobcode_group(+)
				  and    a.deptid = f.deptid
				  and    c.goal_year = d.goal_year
				  and    c.um_entity = f.um_entity
				  
                  and    a.jobcode not in ('0001','0004','0007','0011')
                  and    a.um_zdeptid not like '999%'
                  and    a.um_zdeptid not in ('Z0387','Z0476')
				  and    a.ipedsscode not in ('N')
                  and    a.um_campus in ('#PositionInfo.um_campus#')
                  and    (a.most_recent_payroll = '#MaxMostRecentPayroll.max_most_recent_payroll#'
                             or a.empl_status in ('L', 'P', 'W'))
                  and    a.um_jobcode_group in ('#PositionInfo.um_jobcode_group#')
				  
                  and    c.web_avail = 'TRUE'
                  and    c.campus_code in ('#PositionInfo.um_campus#')
                  and    c.goal_year = '#MaxGoalYear.max_goal_year#'
                  and    c.um_jobcode_grp in ('#PositionInfo.um_jobcode_group#')
				  
				  and    f.um_entity = '#PositionInfo.um_entity#'

                group by a.um_campus_descr,
                         a.um_college_descr,
                         c.um_entity,
                         c.um_entity_descr,
				         a.um_jobcode_group, 
                         a.ipedsscode,
                         a.eeo_jobcode_group,
                         a.um_jobcode_group_descr,
                         e.academic_title,
                         e.non_academic_title,
                         c.fem_avail_tot,
                         c.min_avail_tot,
                         d.disabled,
                         d.veteran

                order by a.um_campus_descr, a.um_college_descr, c.um_entity, a.ipedsscode, a.eeo_jobcode_group
